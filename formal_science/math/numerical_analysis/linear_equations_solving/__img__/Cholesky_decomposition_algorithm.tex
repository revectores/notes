\Function{Cholesky\_decomposition}{$A$, $n$}

\For{$j \gets 1$ to $n$}
	\State $l_{jj} \gets \sqrt{a_{jj} - \displaystyle\sum_{k=1}^{j-1} l_{jk}^2}$
	\For{$i \gets j+1$ to $n$}
		\State $l_{ij} \gets \dfrac{1}{l_{jj}} \left(a_{ij} - \displaystyle\sum_{k=1}^{j-1} l_{ik}l_{jk} \right)$
	\EndFor
\EndFor

\State \Return $L$

\EndFunction
